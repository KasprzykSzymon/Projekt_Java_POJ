public void wyswietlDostepneSamochody(Date dataOd, Date dataDo) {
    System.out.println("\nDostepne samochody dla okresu od " + dataOd + " do " + dataDo + ":");

    for (Samochod samochod : samochody) {
        if (samochod.czyDostepny() && dataOd.compareTo(samochod.getDostepnyOd()) <= 0 && dataDo.compareTo(samochod.getDostepnyOd()) >= 0) {
            System.out.println(samochod);
        }
    }
}
